\href{https://github.com/csstools/postcss-plugins/tree/main/plugins/postcss-gap-properties}{\texttt{ Post\+CSS Gap Properties}} lets you use the {\ttfamily gap}, {\ttfamily column-\/gap}, and {\ttfamily row-\/gap} shorthand properties in CSS, following the \href{https://www.w3.org/TR/css-grid-1/\#gutters}{\texttt{ CSS Grid Layout}} specification.


\begin{DoxyCode}{0}
\DoxyCodeLine{.standard-\/grid \{}
\DoxyCodeLine{    gap: 20px;}
\DoxyCodeLine{\}}
\DoxyCodeLine{}
\DoxyCodeLine{.spaced-\/grid \{}
\DoxyCodeLine{    column-\/gap: 40px;}
\DoxyCodeLine{    row-\/gap: 20px;}
\DoxyCodeLine{\}}
\DoxyCodeLine{}
\DoxyCodeLine{/* becomes */}
\DoxyCodeLine{}
\DoxyCodeLine{.standard-\/grid \{}
\DoxyCodeLine{    grid-\/gap: 20px;}
\DoxyCodeLine{    gap: 20px;}
\DoxyCodeLine{\}}
\DoxyCodeLine{}
\DoxyCodeLine{.spaced-\/grid \{}
\DoxyCodeLine{    grid-\/column-\/gap: 40px;}
\DoxyCodeLine{    column-\/gap: 40px;}

\DoxyCodeLine{    row-\/gap: 20px;}
\DoxyCodeLine{\}}

\end{DoxyCode}
